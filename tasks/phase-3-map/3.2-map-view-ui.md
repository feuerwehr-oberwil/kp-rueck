```typescript
'use client';

import { MapContainer, TileLayer, Marker, Popup, useMap } from 'react-leaflet';
import { Icon, LatLngExpression } from 'leaflet';
import { useEffect } from 'react';
import 'leaflet/dist/leaflet.css';
import { useIncidents, Incident } from '@/lib/contexts/incidents-context';

// Fix Leaflet icon paths in Next.js
import icon from 'leaflet/dist/images/marker-icon.png';
import iconShadow from 'leaflet/dist/images/marker-shadow.png';

const DefaultIcon = new Icon({
  iconUrl: icon.src,
  shadowUrl: iconShadow.src,
  iconSize: [25, 41],
  iconAnchor: [12, 41],
});

// Status-based marker colors
const STATUS_COLORS = {
  eingegangen: '#ef4444',    // red
  reko: '#f97316',           // orange
  disponiert: '#eab308',     // yellow
  einsatz: '#3b82f6',        // blue
  einsatz_beendet: '#22c55e', // green
  abschluss: '#6b7280',      // gray
};

function createColoredIcon(color: string, isTraining: boolean) {
  const svg = `
    <svg width="25" height="41" xmlns="http://www.w3.org/2000/svg">
      <path
        d="M12.5 0C5.6 0 0 5.6 0 12.5c0 9.4 12.5 28.5 12.5 28.5S25 21.9 25 12.5C25 5.6 19.4 0 12.5 0z"
        fill="${color}"
        ${isTraining ? 'stroke="#3b82f6" stroke-width="2" stroke-dasharray="4"' : ''}
      />
      <circle cx="12.5" cy="12.5" r="6" fill="white" />
    </svg>
  `;

  return new Icon({
    iconUrl: `data:image/svg+xml;base64,${btoa(svg)}`,
    iconSize: [25, 41],
    iconAnchor: [12, 41],
    popupAnchor: [0, -41],
  });
}

export function MapView() {
  const { incidents } = useIncidents();

  // Filter incidents with valid coordinates
  const mappableIncidents = incidents.filter(
    i => i.location_lat && i.location_lng
  );

  // Calculate map center (average of all incidents)
  const center: LatLngExpression = mappableIncidents.length > 0
    ? [
        mappableIncidents.reduce((sum, i) => sum + (i.location_lat || 0), 0) / mappableIncidents.length,
        mappableIncidents.reduce((sum, i) => sum + (i.location_lng || 0), 0) / mappableIncidents.length,
      ]
    : [47.5, 7.6]; // Default to Switzerland center

  return (
    <div className="h-full w-full relative">
      <MapContainer
        center={center}
        zoom={13}
        className="h-full w-full"
        zoomControl={true}
      >
        <TileLayer
          attribution='&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
          url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
        />

        {/* Incident Markers */}
        {mappableIncidents.map(incident => (
          <Marker
            key={incident.id}
            position={[incident.location_lat!, incident.location_lng!]}
            icon={createColoredIcon(
              STATUS_COLORS[incident.status],
              incident.training_flag
            )}
          >
            <Popup>
              <IncidentPopup incident={incident} />
            </Popup>
          </Marker>
        ))}

        {/* Auto-fit bounds to show all markers */}
        <FitBounds incidents={mappableIncidents} />
      </MapContainer>

      {/* Warning for incidents without location */}
      {incidents.length !== mappableIncidents.length && (
        <MissingLocationsWarning
          count={incidents.length - mappableIncidents.length}
        />
      )}
    </div>
  );
}

function IncidentPopup({ incident }: { incident: Incident }) {
  return (
    <div className="min-w-[200px]">
      <h3 className="font-bold text-lg mb-2">{incident.title}</h3>

      <div className="space-y-1 text-sm">
        <p>
          <strong>Type:</strong> {incident.type}
        </p>
        <p>
          <strong>Priority:</strong>{' '}
          <span className={`font-bold ${
            incident.priority === 'critical' ? 'text-red-600' :
            incident.priority === 'high' ? 'text-orange-600' :
            'text-gray-600'
          }`}>
            {incident.priority}
          </span>
        </p>
        <p>
          <strong>Status:</strong> {incident.status}
        </p>
        {incident.location_address && (
          <p>
            <strong>Address:</strong> {incident.location_address}
          </p>
        )}
        {incident.training_flag && (
          <p className="text-blue-600 font-bold">Training Mode</p>
        )}
      </div>

      <button
        className="mt-3 w-full bg-blue-500 text-white px-3 py-1 rounded hover:bg-blue-600"
        onClick={() => {
          // Open incident details modal
          window.location.href = `/?incident=${incident.id}`;
        }}
      >
        View Details
      </button>
    </div>
  );
}

function FitBounds({ incidents }: { incidents: Incident[] }) {
  const map = useMap();

  useEffect(() => {
    if (incidents.length === 0) return;

    const bounds = incidents.map(i => [i.location_lat!, i.location_lng!] as LatLngExpression);
    map.fitBounds(bounds, { padding: [50, 50] });
  }, [incidents, map]);

  return null;
}

function MissingLocationsWarning({ count }: { count: number }) {
  return (
    <div className="absolute top-4 left-1/2 transform -translate-x-1/2 bg-yellow-100 border border-yellow-400 text-yellow-800 px-4 py-2 rounded shadow-lg z-[1000]">
      ⚠️ {count} incident{count !== 1 ? 's' : ''} without valid location
    </div>
  );
}
```
