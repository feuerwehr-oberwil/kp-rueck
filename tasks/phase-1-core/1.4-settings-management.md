### 1. Create Settings Service

**File**: `backend/app/services/settings.py`

```python
"""Settings management service."""
from typing import Any
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from ..models import Setting

DEFAULT_SETTINGS = {
    'polling_interval_ms': '5000',
    'training_mode': 'false',
    'auto_archive_timeout_hours': '24',
    'notification_enabled': 'false',
    'alarm_webhook_secret': 'CHANGE_ME_IN_PRODUCTION',
}

async def get_setting(db: AsyncSession, key: str) -> str | None:
    """Get setting value by key."""
    result = await db.execute(select(Setting).where(Setting.key == key))
    setting = result.scalar_one_or_none()
    return setting.value if setting else None

async def get_all_settings(db: AsyncSession) -> dict[str, str]:
    """Get all settings as dict."""
    result = await db.execute(select(Setting))
    settings = result.scalars().all()
    return {s.key: s.value for s in settings}

async def update_setting(
    db: AsyncSession,
    key: str,
    value: str,
    user_id: uuid.UUID
) -> Setting:
    """Update or create setting."""
    result = await db.execute(select(Setting).where(Setting.key == key))
    setting = result.scalar_one_or_none()

    if setting:
        setting.value = value
        setting.updated_by = user_id
        setting.updated_at = datetime.utcnow()
    else:
        setting = Setting(key=key, value=value, updated_by=user_id)
        db.add(setting)

    await db.commit()
    await db.refresh(setting)
    return setting

async def initialize_default_settings(db: AsyncSession):
    """Create default settings if they don't exist."""
    for key, value in DEFAULT_SETTINGS.items():
        existing = await get_setting(db, key)
        if existing is None:
            db.add(Setting(key=key, value=value))
    await db.commit()
```

### 2. Create API Endpoints

**File**: `backend/app/api/settings.py`

```python
"""Settings API endpoints."""
from fastapi import APIRouter, Depends
from sqlalchemy.ext.asyncio import AsyncSession

from .. import schemas
from ..auth.dependencies import CurrentEditor, CurrentUser
from ..database import get_db
from ..services import settings as settings_service

router = APIRouter(prefix="/settings", tags=["settings"])

@router.get("/", response_model=dict[str, str])
async def get_all_settings(
    current_user: CurrentUser = Depends(),
    db: AsyncSession = Depends(get_db)
):
    """Get all settings (any authenticated user)."""
    return await settings_service.get_all_settings(db)

@router.get("/{key}", response_model=schemas.SettingResponse)
async def get_setting(
    key: str,
    current_user: CurrentUser = Depends(),
    db: AsyncSession = Depends(get_db)
):
    """Get single setting."""
    value = await settings_service.get_setting(db, key)
    if not value:
        raise HTTPException(status_code=404, detail="Setting not found")
    return {"key": key, "value": value}

@router.patch("/{key}", response_model=schemas.SettingResponse)
async def update_setting(
    key: str,
    update: schemas.SettingUpdate,
    current_user: CurrentEditor = Depends(),  # Editor only
    db: AsyncSession = Depends(get_db)
):
    """Update setting (editor only)."""
    setting = await settings_service.update_setting(
        db, key, update.value, current_user.id
    )
    return setting
```

### 3. Frontend Settings UI

**File**: `frontend/app/settings/page.tsx`

```typescript
'use client';

import { useState, useEffect } from 'react';
import { useAuth } from '@/lib/contexts/auth-context';

export default function SettingsPage() {
  const { isEditor } = useAuth();
  const [settings, setSettings] = useState<Record<string, string>>({});

  useEffect(() => {
    fetch('/api/settings', { credentials: 'include' })
      .then(res => res.json())
      .then(setSettings);
  }, []);

  const updateSetting = async (key: string, value: string) => {
    await fetch(`/api/settings/${key}`, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ value }),
      credentials: 'include',
    });

    setSettings(prev => ({ ...prev, [key]: value }));
  };

  return (
    <div className="p-4">
      <h1>System Settings</h1>

      <div className="space-y-4">
        <div>
          <label>Polling Interval (ms)</label>
          <input
            type="number"
            value={settings.polling_interval_ms || '5000'}
            onChange={(e) => updateSetting('polling_interval_ms', e.target.value)}
            disabled={!isEditor}
          />
        </div>

        <div>
          <label>Training Mode</label>
          <select
            value={settings.training_mode || 'false'}
            onChange={(e) => updateSetting('training_mode', e.target.value)}
            disabled={!isEditor}
          >
            <option value="false">Live Mode</option>
            <option value="true">Training Mode</option>
          </select>
        </div>
      </div>
    </div>
  );
}
```
