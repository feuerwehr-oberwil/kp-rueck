**File**: `frontend/components/filters/incident-filters.tsx`

**Note**: This is an optional refactoring. The current inline implementation works fine. Only implement if code reuse is needed across multiple pages.

```typescript
'use client'

import { useState } from 'react'
import { Button } from '@/components/ui/button'
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select'
import { Badge } from '@/components/ui/badge'
import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover'
import { Filter, X } from 'lucide-react'

interface FilterValues {
  vehicle: string
  priority: string
  incidentType: string
  training: string
}

interface IncidentFiltersProps {
  filters: FilterValues
  onFilterChange: (filters: FilterValues) => void
  vehicleTypes: Array<{ name: string }>
}

export default function IncidentFilters({
  filters,
  onFilterChange,
  vehicleTypes
}: IncidentFiltersProps) {
  const activeFiltersCount = Object.values(filters).filter(v => v !== 'all').length

  function handleFilterChange(key: keyof FilterValues, value: string) {
    onFilterChange({ ...filters, [key]: value })
  }

  function clearFilters() {
    onFilterChange({
      vehicle: 'all',
      priority: 'all',
      incidentType: 'all',
      training: 'all'
    })
  }

  return (
    <Popover>
      <PopoverTrigger asChild>
        <Button variant="outline" size="sm" className="gap-2">
          <Filter className="h-4 w-4" />
          Filter
          {activeFiltersCount > 0 && (
            <Badge variant="secondary" className="ml-1 h-5 w-5 rounded-full p-0 flex items-center justify-center text-xs">
              {activeFiltersCount}
            </Badge>
          )}
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-80" align="end">
        <div className="space-y-4">
          <h4 className="font-semibold text-sm mb-2">Filter</h4>

          {/* Vehicle filter */}
          <div className="space-y-2">
            <label className="text-xs text-muted-foreground">Fahrzeug</label>
            <Select value={filters.vehicle} onValueChange={(v) => handleFilterChange('vehicle', v)}>
              <SelectTrigger className="h-9">
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">Alle</SelectItem>
                <SelectItem value="none">Keine</SelectItem>
                {vehicleTypes.map((vt) => (
                  <SelectItem key={vt.name} value={vt.name}>
                    {vt.name}
                  </SelectItem>
                ))}
              </SelectContent>
            </Select>
          </div>

          {/* Priority filter */}
          <div className="space-y-2">
            <label className="text-xs text-muted-foreground">Priorität</label>
            <Select value={filters.priority} onValueChange={(v) => handleFilterChange('priority', v)}>
              <SelectTrigger className="h-9">
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">Alle</SelectItem>
                <SelectItem value="high">Hoch</SelectItem>
                <SelectItem value="medium">Mittel</SelectItem>
                <SelectItem value="low">Niedrig</SelectItem>
              </SelectContent>
            </Select>
          </div>

          {/* Incident type filter */}
          <div className="space-y-2">
            <label className="text-xs text-muted-foreground">Einsatzart</label>
            <Select value={filters.incidentType} onValueChange={(v) => handleFilterChange('incidentType', v)}>
              <SelectTrigger className="h-9">
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">Alle</SelectItem>
                {/* Add incident type options */}
              </SelectContent>
            </Select>
          </div>

          {/* Training mode filter */}
          <div className="space-y-2">
            <label className="text-xs text-muted-foreground">Modus</label>
            <Select value={filters.training} onValueChange={(v) => handleFilterChange('training', v)}>
              <SelectTrigger className="h-9">
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">Live + Übung</SelectItem>
                <SelectItem value="false">Nur Live</SelectItem>
                <SelectItem value="true">Nur Übung</SelectItem>
              </SelectContent>
            </Select>
          </div>

          {activeFiltersCount > 0 && (
            <Button variant="ghost" size="sm" className="w-full" onClick={clearFilters}>
              <X className="mr-2 h-4 w-4" />
              Filter zurücksetzen
            </Button>
          )}
        </div>
      </PopoverContent>
    </Popover>
  )
}
```
