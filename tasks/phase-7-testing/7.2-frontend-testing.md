import { render, screen, waitFor } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import { describe, it, expect, vi, beforeEach } from 'vitest'
import PersonnelList from '../personnel-list'
import * as apiClient from '@/lib/api-client'

vi.mock('@/lib/api-client')

describe('PersonnelList', () => {
  const mockPersonnel = [
    { id: '1', name: 'John Doe', role: 'Gruppenführer', availability: 'available' },
    { id: '2', name: 'Jane Smith', role: 'Maschinist', availability: 'assigned' }
  ]

  beforeEach(() => {
    vi.spyOn(apiClient.apiClient, 'getAllPersonnel').mockResolvedValue(mockPersonnel)
  })

  it('loads and displays personnel', async () => {
    render(<PersonnelList />)

    await waitFor(() => {
      expect(screen.getByText('John Doe')).toBeInTheDocument()
      expect(screen.getByText('Jane Smith')).toBeInTheDocument()
    })
  })

  it('filters personnel by search query', async () => {
    render(<PersonnelList />)

    await waitFor(() => screen.getByText('John Doe'))

    const searchInput = screen.getByPlaceholderText(/Search personnel/)
    await userEvent.type(searchInput, 'John')

    expect(screen.getByText('John Doe')).toBeInTheDocument()
    expect(screen.queryByText('Jane Smith')).not.toBeInTheDocument()
  })

  it('opens add dialog', async () => {
    render(<PersonnelList />)

    await userEvent.click(screen.getByText('Add Personnel'))

    expect(screen.getByText(/Add a new team member/)).toBeInTheDocument()
  })
})

### 3. Hook Tests

**File**: `frontend/lib/hooks/__tests__/use-polling.test.ts`

```typescript
import { renderHook, waitFor } from '@testing-library/react'
import { describe, it, expect, vi } from 'vitest'
import usePolling from '../use-polling'

describe('usePolling', () => {
  it('calls callback on mount', async () => {
    const callback = vi.fn().mockResolvedValue(undefined)

    renderHook(() => usePolling(callback, 1000))

    await waitFor(() => {
      expect(callback).toHaveBeenCalledTimes(1)
    })
  })

  it('polls at specified interval', async () => {
    vi.useFakeTimers()
    const callback = vi.fn().mockResolvedValue(undefined)

    renderHook(() => usePolling(callback, 1000))

    await waitFor(() => expect(callback).toHaveBeenCalledTimes(1))

    vi.advanceTimersByTime(1000)
    await waitFor(() => expect(callback).toHaveBeenCalledTimes(2))

    vi.advanceTimersByTime(1000)
    await waitFor(() => expect(callback).toHaveBeenCalledTimes(3))

    vi.useRealTimers()
  })

  it('cleans up interval on unmount', () => {
    const callback = vi.fn()
    const { unmount } = renderHook(() => usePolling(callback, 1000))

    unmount()

    // Callback should not be called after unmount
    setTimeout(() => {
      expect(callback).toHaveBeenCalledTimes(1) // Only initial call
    }, 2000)
  })
})
```

### 4. Context Tests

**File**: `frontend/lib/contexts/__tests__/operations-context.test.tsx`

import { renderHook, act, waitFor } from '@testing-library/react'
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { OperationsProvider, useOperations } from '../operations-context'
import * as apiClient from '@/lib/api-client'

vi.mock('@/lib/api-client')

describe('OperationsContext', () => {
  const wrapper = ({ children }: { children: React.ReactNode }) => (
    <OperationsProvider>{children}</OperationsProvider>
  )

  beforeEach(() => {
    vi.clearAllMocks()
  })

  it('loads incidents on mount', async () => {
    const mockIncidents = [
      { id: '1', title: 'Test 1', status: 'eingegangen' },
      { id: '2', title: 'Test 2', status: 'einsatz' }
    ]

    vi.spyOn(apiClient.apiClient, 'getIncidents').mockResolvedValue(mockIncidents)

    const { result } = renderHook(() => useOperations(), { wrapper })

    await waitFor(() => {
      expect(result.current.incidents).toHaveLength(2)
    })
  })

  it('updates incident status optimistically', async () => {
    const mockIncidents = [
      { id: '1', title: 'Test', status: 'eingegangen' }
    ]

    vi.spyOn(apiClient.apiClient, 'getIncidents').mockResolvedValue(mockIncidents)
    vi.spyOn(apiClient.apiClient, 'updateIncident').mockResolvedValue({
      ...mockIncidents[0],
      status: 'einsatz'
    })

    const { result } = renderHook(() => useOperations(), { wrapper })

    await waitFor(() => expect(result.current.incidents).toHaveLength(1))

    act(() => {
      result.current.updateIncidentStatus('1', 'einsatz')
    })

    // Should update immediately (optimistic)
    expect(result.current.incidents[0].status).toBe('einsatz')
  })
})

### 6. Test Utilities

**File**: `frontend/tests/utils/test-utils.tsx`

```typescript
import { render, RenderOptions } from '@testing-library/react'
import { OperationsProvider } from '@/lib/contexts/operations-context'
import { AuthProvider } from '@/lib/contexts/auth-context'

interface AllProvidersProps {
  children: React.ReactNode
}

function AllProviders({ children }: AllProvidersProps) {
  return (
    <AuthProvider>
      <OperationsProvider>
        {children}
      </OperationsProvider>
    </AuthProvider>
  )
}

function renderWithProviders(
  ui: React.ReactElement,
  options?: Omit<RenderOptions, 'wrapper'>
) {
  return render(ui, { wrapper: AllProviders, ...options })
}

export * from '@testing-library/react'
export { renderWithProviders as render }
```
