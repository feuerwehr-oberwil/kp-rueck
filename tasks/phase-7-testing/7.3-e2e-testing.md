### 3. E2E Test: Complete Incident Workflow

**File**: `frontend/e2e/incident-workflow.spec.ts`

```typescript
import { test, expect } from './fixtures'

test.describe('Complete Incident Workflow', () => {
  test.beforeEach(async ({ page, editorUser }) => {
    // Login as editor
    await page.goto('/login')
    await page.fill('input[name="username"]', editorUser.username)
    await page.fill('input[name="password"]', editorUser.password)
    await page.click('button[type="submit"]')
    await expect(page).toHaveURL('/dashboard')
  })

  test('create incident and move through workflow', async ({ page }) => {
    // Create new incident
    await page.click('text=New Incident')

    await page.fill('input[name="title"]', 'Test Fire Incident')
    await page.selectOption('select[name="type"]', 'fire')
    await page.selectOption('select[name="priority"]', 'high')
    await page.fill('input[name="location"]', 'Hauptstrasse 10, Oberwil')

    await page.click('button:has-text("Create")')

    // Verify incident appears in Eingegangen column
    await expect(page.locator('[data-status="eingegangen"]'))
      .toContainText('Test Fire Incident')

    // Drag to Reko column
    const incident = page.locator('text=Test Fire Incident').first()
    const rekoColumn = page.locator('[data-status="reko"]')

    await incident.dragTo(rekoColumn)

    // Verify moved
    await expect(rekoColumn).toContainText('Test Fire Incident')

    // Assign personnel
    await incident.click()
    await page.click('text=Assign Personnel')
    await page.click('text=John Doe')
    await page.click('button:has-text("Assign")')

    // Verify assignment
    await expect(incident).toContainText('John Doe')

    // Move through remaining statuses
    await incident.dragTo(page.locator('[data-status="disponiert"]'))
    await incident.dragTo(page.locator('[data-status="einsatz"]'))
    await incident.dragTo(page.locator('[data-status="einsatz_beendet"]'))
    await incident.dragTo(page.locator('[data-status="abschluss"]'))

    // Verify incident archived
    await expect(page.locator('[data-status="eingegangen"]'))
      .not.toContainText('Test Fire Incident')
  })
})
```

### 5. E2E Test: Resource Assignment

**File**: `frontend/e2e/resource-assignment.spec.ts`

```typescript
import { test, expect } from './fixtures'

test.describe('Resource Assignment', () => {
  test.beforeEach(async ({ page, editorUser }) => {
    await page.goto('/login')
    await page.fill('input[name="username"]', editorUser.username)
    await page.fill('input[name="password"]', editorUser.password)
    await page.click('button[type="submit"]')
  })

  test('assign and unassign resources', async ({ page }) => {
    // Open incident
    await page.click('.incident-card').first()

    // Assign personnel
    await page.click('text=Assign Personnel')
    await page.click('text=John Doe')
    await page.click('button:has-text("Confirm")')

    await expect(page.locator('.assigned-personnel')).toContainText('John Doe')

    // Assign vehicle
    await page.click('text=Assign Vehicle')
    await page.click('text=TLF 1')
    await page.click('button:has-text("Confirm")')

    await expect(page.locator('.assigned-vehicles')).toContainText('TLF 1')

    // Verify resource status changed
    await page.goto('/resources')
    await expect(page.locator('text=John Doe').locator('..')).toContainText('assigned')

    // Unassign personnel
    await page.goto('/dashboard')
    await page.click('.incident-card').first()
    await page.click('.assigned-personnel >> text=John Doe >> .. >> button')

    await expect(page.locator('.assigned-personnel')).not.toContainText('John Doe')
  })

  test('warns on double-booking', async ({ page }) => {
    // Assign John Doe to first incident
    await page.click('.incident-card').first()
    await page.click('text=Assign Personnel')
    await page.click('text=John Doe')
    await page.click('button:has-text("Confirm")')

    // Try to assign John Doe to second incident
    await page.click('.incident-card').nth(1)
    await page.click('text=Assign Personnel')
    await page.click('text=John Doe')

    // Verify warning shown
    await expect(page.locator('text=already assigned')).toBeVisible()
  })
})
```

### 6. E2E Test: Training Mode

**File**: `frontend/e2e/training-mode.spec.ts`

```typescript
import { test, expect } from './fixtures'

test.describe('Training Mode', () => {
  test.beforeEach(async ({ page, editorUser }) => {
    await page.goto('/login')
    await page.fill('input[name="username"]', editorUser.username)
    await page.fill('input[name="password"]', editorUser.password)
    await page.click('button[type="submit"]')
  })

  test('toggle training mode', async ({ page }) => {
    // Go to settings
    await page.goto('/settings')

    // Enable training mode
    await page.click('input[name="training-mode"]')

    // Verify banner appears
    await expect(page.locator('text=ÜBUNGSMODUS')).toBeVisible()

    // Create incident
    await page.goto('/dashboard')
    await page.click('text=New Incident')
    await page.fill('input[name="title"]', 'Training Incident')
    await page.click('button:has-text("Create")')

    // Verify training badge
    await expect(page.locator('.incident-card').first()).toContainText('Übung')

    // Disable training mode
    await page.goto('/settings')
    await page.click('input[name="training-mode"]')

    // Verify banner gone
    await expect(page.locator('text=ÜBUNGSMODUS')).not.toBeVisible()
  })

  test('filter training incidents', async ({ page }) => {
    await page.goto('/dashboard')

    // Create live incident
    await page.click('text=New Incident')
    await page.fill('input[name="title"]', 'Live Incident')
    await page.click('button:has-text("Create")')

    // Enable training mode and create training incident
    await page.goto('/settings')
    await page.click('input[name="training-mode"]')
    await page.goto('/dashboard')
    await page.click('text=New Incident')
    await page.fill('input[name="title"]', 'Training Incident')
    await page.click('button:has-text("Create")')

    // Filter to live only
    await page.selectOption('select[name="training-filter"]', 'false')

    await expect(page.locator('.incident-card')).toContainText('Live Incident')
    await expect(page.locator('.incident-card')).not.toContainText('Training Incident')

    // Filter to training only
    await page.selectOption('select[name="training-filter"]', 'true')

    await expect(page.locator('.incident-card')).toContainText('Training Incident')
    await expect(page.locator('.incident-card')).not.toContainText('Live Incident')
  })
})
```

import { test, expect, devices } from '@playwright/test'

test.use({ ...devices['iPhone 12'] })

test.describe('Mobile Viewer', () => {
  test('displays incidents on mobile', async ({ page }) => {
    await page.goto('/login')
    await page.fill('input[name="username"]', 'viewer@test.com')
    await page.fill('input[name="password"]', 'test123')
    await page.click('button[type="submit"]')

    // Verify mobile layout
    await expect(page.locator('.mobile-incident-list')).toBeVisible()

    // Verify read-only (no edit buttons)
    await expect(page.locator('button:has-text("Edit")')).not.toBeVisible()
  })

  test('auto-refreshes data', async ({ page }) => {
    await page.goto('/login')
    await page.fill('input[name="username"]', 'viewer@test.com')
    await page.fill('input[name="password"]', 'test123')
    await page.click('button[type="submit"]')

    // Record initial incident count
    const initialCount = await page.locator('.incident-card').count()

    // Wait for auto-refresh (5 seconds)
    await page.waitForTimeout(6000)

    // Count should be same or changed (if data changed)
    const newCount = await page.locator('.incident-card').count()
    expect(typeof newCount).toBe('number')
  })
})

### 8. Test Helpers

**File**: `frontend/e2e/helpers.ts`

import { Page } from '@playwright/test'

export async function loginAsEditor(page: Page) {
  await page.goto('/login')
  await page.fill('input[name="username"]', 'editor@test.com')
  await page.fill('input[name="password"]', 'test123')
  await page.click('button[type="submit"]')
  await page.waitForURL('/dashboard')
}

export async function createIncident(
  page: Page,
  data: { title: string, type: string, priority: string }
) {
  await page.click('text=New Incident')
  await page.fill('input[name="title"]', data.title)
  await page.selectOption('select[name="type"]', data.type)
  await page.selectOption('select[name="priority"]', data.priority)
  await page.click('button:has-text("Create")')
}

export async function waitForToast(page: Page, message: string) {
  await page.waitForSelector(`text=${message}`, { timeout: 5000 })
}
