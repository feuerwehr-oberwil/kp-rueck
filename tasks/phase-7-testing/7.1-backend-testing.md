"""Test database models."""
import pytest
from sqlalchemy.exc import IntegrityError

from app.models import User, Incident, Vehicle


@pytest.mark.asyncio
async def test_user_creation(db_session):
    """Test user model creation."""
    user = User(
        username="testuser",
        password_hash="hashed_password",
        role="editor"
    )
    db_session.add(user)
    await db_session.commit()

    assert user.id is not None
    assert user.username == "testuser"
    assert user.role == "editor"


@pytest.mark.asyncio
async def test_user_unique_username(db_session, test_user):
    """Test username uniqueness constraint."""
    duplicate = User(
        username="testuser",  # Same as test_user
        password_hash="hash",
        role="viewer"
    )
    db_session.add(duplicate)

    with pytest.raises(IntegrityError):
        await db_session.commit()


@pytest.mark.asyncio
async def test_incident_location_constraint(db_session):
    """Test lat/lng must both be present or both null."""
    # Invalid: only lat provided
    incident = Incident(
        title="Test",
        type="fire",
        priority="high",
        status="eingegangen",
        location_lat=47.123,
        location_lng=None  # Missing
    )
    db_session.add(incident)

    with pytest.raises(IntegrityError):
        await db_session.commit()

"""Test incident API endpoints."""
import pytest


@pytest.mark.asyncio
async def test_create_incident(authenticated_client):
    """Test incident creation."""
    response = await authenticated_client.post("/api/incidents", json={
        "title": "Fire at Main St",
        "type": "fire",
        "priority": "high",
        "status": "eingegangen",
        "location_address": "Main Street 123",
        "training_flag": False,
    })

    assert response.status_code == 201
    data = response.json()
    assert data["title"] == "Fire at Main St"
    assert data["type"] == "fire"


@pytest.mark.asyncio
async def test_list_incidents(authenticated_client, test_user):
    """Test listing incidents with filters."""
    # Create test incidents
    await authenticated_client.post("/api/incidents", json={
        "title": "Live Incident",
        "type": "fire",
        "priority": "high",
        "training_flag": False,
    })

    await authenticated_client.post("/api/incidents", json={
        "title": "Training Incident",
        "type": "fire",
        "priority": "low",
        "training_flag": True,
    })

    # List all
    response = await authenticated_client.get("/api/incidents")
    assert response.status_code == 200
    assert len(response.json()) == 2

    # Filter training only
    response = await authenticated_client.get("/api/incidents?training_only=true")
    assert len(response.json()) == 1
    assert response.json()[0]["title"] == "Training Incident"


@pytest.mark.asyncio
async def test_update_incident_status(authenticated_client):
    """Test status update creates transition."""
    # Create incident
    create_resp = await authenticated_client.post("/api/incidents", json={
        "title": "Test",
        "type": "fire",
        "priority": "high",
    })
    incident_id = create_resp.json()["id"]

    # Update status
    response = await authenticated_client.post(
        f"/api/incidents/{incident_id}/status",
        json={
            "from_status": "eingegangen",
            "to_status": "reko",
        }
    )

    assert response.status_code == 200
    assert response.json()["status"] == "reko"

    # Check history
    history_resp = await authenticated_client.get(
        f"/api/incidents/{incident_id}/history"
    )
    assert len(history_resp.json()) == 1


@pytest.mark.asyncio
async def test_viewer_cannot_create(client, viewer_user):
    """Test viewers cannot create incidents."""
    # Login as viewer
    await client.post("/api/auth/login", data={
        "username": "viewer",
        "password": "viewerpass",
    })

    response = await client.post("/api/incidents", json={
        "title": "Test",
        "type": "fire",
        "priority": "high",
    })

    assert response.status_code == 403  # Forbidden

"""Security-focused tests."""


@pytest.mark.asyncio
async def test_password_not_in_response(authenticated_client):
    """Ensure password_hash never in API responses."""
    response = await authenticated_client.get("/api/auth/me")

    data = response.json()
    assert "password_hash" not in data
    assert "password" not in data


@pytest.mark.asyncio
async def test_sql_injection_protection(authenticated_client):
    """Test SQL injection attempts are blocked."""
    # Try SQL injection in title
    response = await authenticated_client.post("/api/incidents", json={
        "title": "'; DROP TABLE incidents; --",
        "type": "fire",
        "priority": "high",
    })

    # Should succeed but title is escaped
    assert response.status_code == 201
    assert response.json()["title"] == "'; DROP TABLE incidents; --"

    # Verify table still exists
    list_resp = await authenticated_client.get("/api/incidents")
    assert list_resp.status_code == 200
