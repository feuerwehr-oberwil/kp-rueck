### 1. DIVERA API Client

**File**: `backend/app/services/divera_client.py`

```python
"""
DIVERA 24/7 API client.

Documentation: https://api.divera247.com/
"""

import logging
import aiohttp
import asyncio
from typing import Optional, List, Dict, Any
from datetime import datetime, timedelta
from enum import Enum

from app.config import get_settings

logger = logging.getLogger(__name__)


class DiveraAlarmStatus(Enum):
    """DIVERA alarm status values."""
    NEW = 1
    ACKNOWLEDGED = 2
    IN_PROGRESS = 3
    COMPLETED = 4
    CANCELLED = 5


class DiveraClient:
    """Client for DIVERA 24/7 API."""

    def __init__(self):
        self.settings = get_settings()
        self.api_key = self.settings.DIVERA_API_KEY
        self.base_url = "https://api.divera247.com/v2"
        self.session: Optional[aiohttp.ClientSession] = None

        # Rate limiting (max 60 req/min for DIVERA API)
        self._last_request_time = 0.0
        self._rate_limit_delay = 1.0  # 1 second between requests

    async def __aenter__(self):
        """Context manager entry."""
        self.session = aiohttp.ClientSession(
            headers={
                "Authorization": f"Bearer {self.api_key}",
                "Content-Type": "application/json",
                "User-Agent": "KP-Rueck-Feuerwehr-Oberwil/1.0"
            },
            timeout=aiohttp.ClientTimeout(total=30)
        )
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """Context manager exit."""
        if self.session:
            await self.session.close()

    async def _rate_limit(self):
        """Enforce rate limiting."""
        now = asyncio.get_event_loop().time()
        time_since_last = now - self._last_request_time
        if time_since_last < self._rate_limit_delay:
            await asyncio.sleep(self._rate_limit_delay - time_since_last)
        self._last_request_time = asyncio.get_event_loop().time()

    async def _request(
        self,
        method: str,
        endpoint: str,
        **kwargs
    ) -> Dict[str, Any]:
        """Make API request with rate limiting and error handling."""
        if not self.session:
            raise RuntimeError("Client not initialized. Use 'async with' context.")

        await self._rate_limit()

        url = f"{self.base_url}{endpoint}"

        try:
            async with self.session.request(method, url, **kwargs) as response:
                response.raise_for_status()
                data = await response.json()
                return data

        except aiohttp.ClientResponseError as e:
            logger.error(f"DIVERA API error: {e.status} - {e.message}")
            raise
        except asyncio.TimeoutError:
            logger.error(f"DIVERA API timeout: {endpoint}")
            raise
        except Exception as e:
            logger.error(f"DIVERA API request failed: {str(e)}")
            raise

    async def get_alarms(
        self,
        since: Optional[datetime] = None,
        limit: int = 50
    ) -> List[Dict[str, Any]]:
        """
        Get recent alarms from DIVERA.

        Args:
            since: Get alarms since this timestamp (default: last 24 hours)
            limit: Maximum number of alarms to return

        Returns:
            List of alarm objects
        """
        if not since:
            since = datetime.now() - timedelta(hours=24)

        params = {
            "since": int(since.timestamp()),
            "limit": limit
        }

        data = await self._request("GET", "/alarms", params=params)
        return data.get("data", [])

    async def get_alarm_details(self, alarm_id: str) -> Dict[str, Any]:
        """Get detailed information for specific alarm."""
        data = await self._request("GET", f"/alarms/{alarm_id}")
        return data.get("data", {})

    async def update_alarm_status(
        self,
        alarm_id: str,
        status: DiveraAlarmStatus,
        message: Optional[str] = None
    ) -> bool:
        """
        Update alarm status in DIVERA.

        Args:
            alarm_id: DIVERA alarm ID
            status: New status
            message: Optional status message

        Returns:
            True if successful
        """
        payload = {
            "status": status.value,
            "message": message
        }

        try:
            await self._request("PATCH", f"/alarms/{alarm_id}/status", json=payload)
            logger.info(f"Updated DIVERA alarm {alarm_id} status to {status.name}")
            return True
        except Exception as e:
            logger.error(f"Failed to update DIVERA alarm status: {str(e)}")
            return False

    async def send_notification(
        self,
        user_ids: List[str],
        title: str,
        message: str,
        priority: int = 1
    ) -> bool:
        """
        Send notification to specific users via DIVERA.

        Args:
            user_ids: List of DIVERA user IDs
            title: Notification title
            message: Notification message
            priority: Priority level (1-5)

        Returns:
            True if successful
        """
        payload = {
            "users": user_ids,
            "title": title,
            "text": message,
            "priority": priority,
            "vehicle": None,  # Optional: specific vehicle notification
            "group": None     # Optional: specific group notification
        }

        try:
            await self._request("POST", "/messages", json=payload)
            logger.info(f"Sent DIVERA notification to {len(user_ids)} users")
            return True
        except Exception as e:
            logger.error(f"Failed to send DIVERA notification: {str(e)}")
            return False

    async def get_available_personnel(self) -> List[Dict[str, Any]]:
        """Get list of available personnel from DIVERA."""
        data = await self._request("GET", "/users/available")
        return data.get("data", [])

    async def update_incident_details(
        self,
        alarm_id: str,
        assigned_personnel: List[str],
        assigned_vehicles: List[str],
        materials: List[str],
        notes: Optional[str] = None
    ) -> bool:
        """
        Push detailed incident information back to DIVERA.

        Args:
            alarm_id: DIVERA alarm ID
            assigned_personnel: List of personnel names/IDs
            assigned_vehicles: List of vehicle names
            materials: List of materials
            notes: Additional notes

        Returns:
            True if successful
        """
        payload = {
            "personnel": assigned_personnel,
            "vehicles": assigned_vehicles,
            "equipment": materials,
            "notes": notes
        }

        try:
            await self._request("POST", f"/alarms/{alarm_id}/resources", json=payload)
            logger.info(f"Updated DIVERA alarm {alarm_id} with resource details")
            return True
        except Exception as e:
            logger.error(f"Failed to update DIVERA incident details: {str(e)}")
            return False


# Global client instance
_divera_client: Optional[DiveraClient] = None


def get_divera_client() -> DiveraClient:
    """Get DIVERA client instance."""
    global _divera_client
    if _divera_client is None:
        _divera_client = DiveraClient()
    return _divera_client
```

### 2. DIVERA Sync Service

**File**: `backend/app/services/divera_sync.py`

```python
"""
DIVERA synchronization service.
"""

import logging
from typing import List, Dict, Any
from datetime import datetime, timedelta
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select

from app import models, crud
from app.services.divera_client import DiveraClient, DiveraAlarmStatus, get_divera_client
from app.config import get_settings

logger = logging.getLogger(__name__)


class DiveraSyncService:
    """Service for synchronizing data with DIVERA."""

    def __init__(self):
        self.settings = get_settings()
        self.last_sync_time: datetime = datetime.now() - timedelta(hours=24)

    async def poll_new_alarms(self, db: AsyncSession) -> List[models.Incident]:
        """
        Poll DIVERA for new alarms and create incidents.

        Returns:
            List of created incidents
        """
        if not self.settings.DIVERA_INTEGRATION_ENABLED:
            logger.debug("DIVERA integration disabled")
            return []

        created_incidents = []

        try:
            async with get_divera_client() as client:
                # Get alarms since last sync
                alarms = await client.get_alarms(since=self.last_sync_time)

                logger.info(f"Found {len(alarms)} new alarms from DIVERA")

                for alarm in alarms:
                    # Check if incident already exists (by DIVERA alarm ID)
                    existing = await self._find_incident_by_divera_id(
                        db,
                        alarm["id"]
                    )

                    if existing:
                        logger.debug(f"Incident already exists for alarm {alarm['id']}")
                        continue

                    # Create incident from alarm
                    incident = await self._create_incident_from_alarm(db, alarm)
                    if incident:
                        created_incidents.append(incident)

                        # Update DIVERA alarm status to acknowledged
                        await client.update_alarm_status(
                            alarm["id"],
                            DiveraAlarmStatus.ACKNOWLEDGED,
                            f"Incident created: {incident.id}"
                        )

                self.last_sync_time = datetime.now()

        except Exception as e:
            logger.error(f"DIVERA alarm polling failed: {str(e)}")

        return created_incidents

    async def push_incident_details(
        self,
        db: AsyncSession,
        incident: models.Incident
    ) -> bool:
        """
        Push incident details back to DIVERA.

        Args:
            incident: Incident to sync

        Returns:
            True if successful
        """
        if not self.settings.DIVERA_INTEGRATION_ENABLED:
            return False

        # Extract DIVERA alarm ID from incident description
        divera_alarm_id = self._extract_divera_id(incident.description)
        if not divera_alarm_id:
            logger.warning(f"No DIVERA alarm ID found for incident {incident.id}")
            return False

        try:
            # Get assigned resources
            from sqlalchemy import and_
            stmt = select(models.IncidentAssignment).where(
                and_(
                    models.IncidentAssignment.incident_id == incident.id,
                    models.IncidentAssignment.unassigned_at.is_(None)
                )
            )
            result = await db.execute(stmt)
            assignments = result.scalars().all()

            # Group by resource type
            personnel = []
            vehicles = []
            materials = []

            for assignment in assignments:
                if assignment.resource_type == "personnel":
                    person = await db.get(models.Personnel, assignment.resource_id)
                    if person:
                        personnel.append(person.name)
                elif assignment.resource_type == "vehicle":
                    vehicle = await db.get(models.Vehicle, assignment.resource_id)
                    if vehicle:
                        vehicles.append(vehicle.name)
                elif assignment.resource_type == "material":
                    material = await db.get(models.Material, assignment.resource_id)
                    if material:
                        materials.append(material.name)

            # Push to DIVERA
            async with get_divera_client() as client:
                success = await client.update_incident_details(
                    divera_alarm_id,
                    personnel,
                    vehicles,
                    materials,
                    notes=f"Status: {incident.status}"
                )

                if success:
                    logger.info(f"Pushed details for incident {incident.id} to DIVERA")

                return success

        except Exception as e:
            logger.error(f"Failed to push incident details to DIVERA: {str(e)}")
            return False

    async def notify_personnel(
        self,
        user_ids: List[str],
        incident: models.Incident
    ) -> bool:
        """
        Send notification to specific personnel via DIVERA.

        Args:
            user_ids: DIVERA user IDs
            incident: Incident to notify about

        Returns:
            True if successful
        """
        if not self.settings.DIVERA_INTEGRATION_ENABLED:
            logger.info(f"[DUMMY] Would notify DIVERA users: {user_ids}")
            return True

        try:
            async with get_divera_client() as client:
                title = f"Einsatz: {incident.type.upper()}"
                message = f"{incident.title}\n{incident.location_address}\nPriorität: {incident.priority}"

                success = await client.send_notification(
                    user_ids,
                    title,
                    message,
                    priority=3 if incident.priority in ["high", "critical"] else 1
                )

                return success

        except Exception as e:
            logger.error(f"Failed to send DIVERA notification: {str(e)}")
            return False

    async def _find_incident_by_divera_id(
        self,
        db: AsyncSession,
        divera_id: str
    ) -> Optional[models.Incident]:
        """Find incident by DIVERA alarm ID in description."""
        stmt = select(models.Incident).where(
            models.Incident.description.like(f"%DIVERA#{divera_id}%")
        )
        result = await db.execute(stmt)
        return result.scalar_one_or_none()

    def _extract_divera_id(self, description: str) -> Optional[str]:
        """Extract DIVERA alarm ID from incident description."""
        import re
        match = re.search(r"DIVERA#(\w+)", description)
        return match.group(1) if match else None

    async def _create_incident_from_alarm(
        self,
        db: AsyncSession,
        alarm: Dict[str, Any]
    ) -> Optional[models.Incident]:
        """Create incident from DIVERA alarm data."""
        from app import schemas

        try:
            incident_data = schemas.IncidentCreate(
                title=f"{alarm.get('title', 'DIVERA Alarm')}",
                type=self._map_alarm_type(alarm.get('type')),
                priority=self._map_alarm_priority(alarm.get('priority', 1)),
                location_address=alarm.get('address', ''),
                location_lat=alarm.get('lat'),
                location_lng=alarm.get('lng'),
                description=f"DIVERA#{alarm['id']}\n\n{alarm.get('text', '')}",
                training_flag=False,
                status="eingegangen"
            )

            # Get first editor user for creation
            stmt = select(models.User).where(models.User.role == "editor").limit(1)
            result = await db.execute(stmt)
            user = result.scalar_one_or_none()

            if not user:
                logger.error("No editor user found")
                return None

            incident = await crud.create_incident(db, incident_data, user.id)
            logger.info(f"Created incident {incident.id} from DIVERA alarm {alarm['id']}")

            return incident

        except Exception as e:
            logger.error(f"Failed to create incident from DIVERA alarm: {str(e)}")
            return None

    def _map_alarm_type(self, divera_type: Optional[str]) -> str:
        """Map DIVERA alarm type to incident type."""
        mapping = {
            "fire": "fire",
            "medical": "medical",
            "technical": "technical",
            "hazmat": "hazmat"
        }
        return mapping.get(divera_type, "other")

    def _map_alarm_priority(self, divera_priority: int) -> str:
        """Map DIVERA priority (1-5) to incident priority."""
        if divera_priority >= 4:
            return "critical"
        elif divera_priority == 3:
            return "high"
        elif divera_priority == 2:
            return "medium"
        else:
            return "low"


# Global service instance
_divera_sync: Optional[DiveraSyncService] = None


def get_divera_sync_service() -> DiveraSyncService:
    """Get DIVERA sync service instance."""
    global _divera_sync
    if _divera_sync is None:
        _divera_sync = DiveraSyncService()
    return _divera_sync
```

### 3. Background Polling Task

**File**: `backend/app/tasks/divera_poll.py`

```python
"""
Background task to poll DIVERA for new alarms.
"""

import asyncio
import logging
from sqlalchemy.ext.asyncio import AsyncSession

from app.database import AsyncSessionLocal
from app.services.divera_sync import get_divera_sync_service
from app.config import get_settings

logger = logging.getLogger(__name__)


async def poll_divera_alarms():
    """Poll DIVERA for new alarms (run periodically)."""
    settings = get_settings()

    if not settings.DIVERA_INTEGRATION_ENABLED:
        logger.debug("DIVERA integration disabled, skipping poll")
        return

    logger.info("Polling DIVERA for new alarms...")

    async with AsyncSessionLocal() as db:
        try:
            sync_service = get_divera_sync_service()
            incidents = await sync_service.poll_new_alarms(db)

            logger.info(f"Created {len(incidents)} incidents from DIVERA alarms")

        except Exception as e:
            logger.error(f"DIVERA polling failed: {str(e)}", exc_info=True)
        finally:
            await db.close()


async def start_divera_polling():
    """Start background polling task."""
    settings = get_settings()

    if not settings.DIVERA_INTEGRATION_ENABLED:
        logger.info("DIVERA integration disabled")
        return

    poll_interval = settings.DIVERA_POLL_INTERVAL_SECONDS

    logger.info(f"Starting DIVERA polling (interval: {poll_interval}s)")

    while True:
        try:
            await poll_divera_alarms()
        except Exception as e:
            logger.error(f"DIVERA polling error: {str(e)}")

        await asyncio.sleep(poll_interval)
```

```python
"""
DIVERA integration API endpoints.
"""

from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from typing import List

from app.database import get_db
from app.models import User
from app.auth import require_role
from app.services.divera_sync import get_divera_sync_service
from app import models

router = APIRouter(prefix="/divera", tags=["divera"])


@router.post("/sync")
async def trigger_divera_sync(
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(require_role("editor"))
):
    """
    Manually trigger DIVERA alarm sync.

    Editor role required.
    """
    sync_service = get_divera_sync_service()
    incidents = await sync_service.poll_new_alarms(db)

    return {
        "message": f"Created {len(incidents)} incidents from DIVERA",
        "incidents": [str(i.id) for i in incidents]
    }


@router.post("/push/{incident_id}")
async def push_incident_to_divera(
    incident_id: str,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(require_role("editor"))
):
    """
    Push incident details to DIVERA.

    Editor role required.
    """
    incident = await db.get(models.Incident, incident_id)
    if not incident:
        raise HTTPException(status_code=404, detail="Incident not found")

    sync_service = get_divera_sync_service()
    success = await sync_service.push_incident_details(db, incident)

    if not success:
        raise HTTPException(status_code=500, detail="Failed to push to DIVERA")

    return {"message": "Incident details pushed to DIVERA"}


@router.post("/notify")
async def send_divera_notification(
    user_ids: List[str],
    incident_id: str,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(require_role("editor"))
):
    """
    Send notification to specific personnel via DIVERA.

    Editor role required.
    """
    incident = await db.get(models.Incident, incident_id)
    if not incident:
        raise HTTPException(status_code=404, detail="Incident not found")

    sync_service = get_divera_sync_service()
    success = await sync_service.notify_personnel(user_ids, incident)

    if not success:
        raise HTTPException(status_code=500, detail="Failed to send notification")

    return {
        "message": f"Notification sent to {len(user_ids)} users",
        "user_ids": user_ids
    }
```
