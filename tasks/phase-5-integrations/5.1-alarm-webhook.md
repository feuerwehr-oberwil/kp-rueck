"""
Alarm webhook service for processing incoming alarms.
"""

import logging
import hashlib
import hmac
from typing import Optional, Dict, Any
from datetime import datetime
from sqlalchemy.ext.asyncio import AsyncSession

from app import models, schemas, crud
from app.services.geocoding import get_geocoding_service
from app.config import get_settings

logger = logging.getLogger(__name__)


class AlarmWebhookService:
    """Service for processing alarm webhooks."""

    # Type mapping from alarm server to incident types
    TYPE_MAPPING = {
        "feuer": "fire",
        "brand": "fire",
        "fire": "fire",
        "medical": "medical",
        "medizinisch": "medical",
        "rettung": "medical",
        "technical": "technical",
        "technisch": "technical",
        "hilfeleistung": "technical",
        "hazmat": "hazmat",
        "gefahrgut": "hazmat",
        "chemie": "hazmat",
    }

    # Priority mapping based on keywords
    PRIORITY_KEYWORDS = {
        "critical": ["person in gefahr", "menschenleben", "explosion", "gross", "mehrere verletzte"],
        "high": ["feuer", "brand", "rauch", "verletzte", "unfall"],
        "medium": ["klein", "kontrolle", "verdacht"],
        "low": ["false alarm", "fehlalarm", "übung"]
    }

    def __init__(self):
        self.settings = get_settings()

    def verify_signature(self, payload: bytes, signature: str) -> bool:
        """
        Verify webhook signature using HMAC-SHA256.

        Args:
            payload: Raw request body
            signature: Signature from X-Webhook-Signature header

        Returns:
            True if signature is valid
        """
        secret = self.settings.ALARM_WEBHOOK_SECRET.encode()
        expected_signature = hmac.new(
            secret,
            payload,
            hashlib.sha256
        ).hexdigest()

        return hmac.compare_digest(signature, expected_signature)

    def map_incident_type(self, alarm_type: str, description: str) -> str:
        """
        Map alarm type to incident type.

        Args:
            alarm_type: Type from alarm payload
            description: Alarm description for keyword matching

        Returns:
            Incident type (fire, medical, technical, hazmat, other)
        """
        # Check direct type mapping
        alarm_type_lower = alarm_type.lower()
        if alarm_type_lower in self.TYPE_MAPPING:
            return self.TYPE_MAPPING[alarm_type_lower]

        # Check description for keywords
        description_lower = description.lower()
        for keyword, incident_type in self.TYPE_MAPPING.items():
            if keyword in description_lower:
                return incident_type

        # Default to other
        return "other"

    def determine_priority(self, description: str, alarm_type: str) -> str:
        """
        Determine incident priority based on keywords.

        Args:
            description: Alarm description
            alarm_type: Alarm type

        Returns:
            Priority (low, medium, high, critical)
        """
        description_lower = (description + " " + alarm_type).lower()

        # Check keywords for each priority level
        for priority in ["critical", "high", "medium", "low"]:
            keywords = self.PRIORITY_KEYWORDS.get(priority, [])
            if any(keyword in description_lower for keyword in keywords):
                return priority

        # Default to medium
        return "medium"

    async def check_duplicate(
        self,
        db: AsyncSession,
        alarm_id: str,
        location_address: str,
        time_window_minutes: int = 30
    ) -> Optional[models.Incident]:
        """
        Check if similar incident already exists (duplicate detection).

        Args:
            db: Database session
            alarm_id: External alarm ID
            location_address: Alarm location
            time_window_minutes: Time window for duplicate detection

        Returns:
            Existing incident if found, None otherwise
        """
        # Check by alarm_id in description (if we stored it)
        # Or by location + recent timestamp

        from sqlalchemy import select, and_
        from datetime import timedelta

        cutoff_time = datetime.now() - timedelta(minutes=time_window_minutes)

        stmt = select(models.Incident).where(
            and_(
                models.Incident.location_address.ilike(f"%{location_address}%"),
                models.Incident.created_at >= cutoff_time,
                models.Incident.status != "abschluss"  # Not archived
            )
        )

        result = await db.execute(stmt)
        existing_incident = result.scalar_one_or_none()

        if existing_incident:
            logger.warning(
                f"Duplicate alarm detected: alarm_id={alarm_id}, "
                f"existing_incident={existing_incident.id}"
            )

        return existing_incident

    async def process_alarm(
        self,
        db: AsyncSession,
        alarm_data: Dict[str, Any]
    ) -> models.Incident:
        """
        Process incoming alarm and create incident.

        Args:
            db: Database session
            alarm_data: Parsed alarm payload

        Returns:
            Created (or existing) incident
        """
        # Extract alarm data
        alarm_id = alarm_data.get("alarm_id")
        timestamp = alarm_data.get("timestamp")
        alarm_type = alarm_data.get("type", "other")
        alarm_priority = alarm_data.get("priority")
        description = alarm_data.get("description", "")
        location = alarm_data.get("location", {})
        source = alarm_data.get("source", "alarm_webhook")

        location_address = location.get("address", "")
        location_lat = location.get("lat")
        location_lng = location.get("lng")

        # Check for duplicates
        if location_address:
            existing = await self.check_duplicate(db, alarm_id, location_address)
            if existing:
                logger.info(f"Returning existing incident: {existing.id}")
                return existing

        # Map incident type
        incident_type = self.map_incident_type(alarm_type, description)

        # Determine priority (use alarm priority if provided, otherwise determine)
        if alarm_priority and alarm_priority in ["low", "medium", "high", "critical"]:
            priority = alarm_priority
        else:
            priority = self.determine_priority(description, alarm_type)

        # Geocode address if coordinates not provided
        if location_address and (not location_lat or not location_lng):
            geocoding = get_geocoding_service()
            geo_result = await geocoding.geocode_address(location_address)

            if geo_result:
                location_lat = geo_result["lat"]
                location_lng = geo_result["lng"]
                location_address = geo_result["display_name"]
                logger.info(f"Geocoded address: {location_address}")
            else:
                logger.warning(f"Failed to geocode address: {location_address}")

        # Create incident title
        title = f"{alarm_type.upper()}: {location_address[:50]}"

        # Build incident data
        incident_data = schemas.IncidentCreate(
            title=title,
            type=incident_type,
            priority=priority,
            location_address=location_address,
            location_lat=location_lat,
            location_lng=location_lng,
            description=f"Alarm #{alarm_id}\n\n{description}\n\nQuelle: {source}",
            training_flag=False,  # Real alarms are never training
            status="eingegangen"
        )

        # Create incident (use system user or create dedicated alarm user)
        # For MVP, use first editor user
        from sqlalchemy import select
        stmt = select(models.User).where(models.User.role == "editor").limit(1)
        result = await db.execute(stmt)
        alarm_user = result.scalar_one_or_none()

        if not alarm_user:
            raise ValueError("No editor user found for alarm processing")

        incident = await crud.create_incident(db, incident_data, alarm_user.id)

        logger.info(
            f"Created incident from alarm: incident_id={incident.id}, "
            f"alarm_id={alarm_id}, type={incident_type}, priority={priority}"
        )

        # Trigger notification (dummy for MVP)
        await self.send_notification(incident, alarm_data)

        return incident

    async def send_notification(
        self,
        incident: models.Incident,
        alarm_data: Dict[str, Any]
    ):
        """
        Send notification about new incident (dummy for MVP).

        In production, this would send to DIVERA or other notification system.
        """
        notification_message = (
            f"New incident created from alarm:\n"
            f"Type: {incident.type}\n"
            f"Priority: {incident.priority}\n"
            f"Location: {incident.location_address}\n"
            f"Alarm ID: {alarm_data.get('alarm_id')}\n"
        )

        # MVP: Just log the notification
        logger.info(f"[NOTIFICATION] Would send to DIVERA:\n{notification_message}")

        # TODO: Implement real DIVERA notification in Task 5.2


# Global service instance
_alarm_webhook: Optional[AlarmWebhookService] = None


def get_alarm_webhook_service() -> AlarmWebhookService:
    """Get or create the global alarm webhook service instance."""
    global _alarm_webhook
    if _alarm_webhook is None:
        _alarm_webhook = AlarmWebhookService()
    return _alarm_webhook

"""
Alarm webhook API endpoints.
"""

from fastapi import APIRouter, HTTPException, Header, Request, Depends
from sqlalchemy.ext.asyncio import AsyncSession
from pydantic import BaseModel, Field
from typing import Optional
import logging

from app.database import get_db
from app.services.alarm_webhook import get_alarm_webhook_service
from app import schemas

router = APIRouter(prefix="/alarms", tags=["alarms"])
logger = logging.getLogger(__name__)


class AlarmLocation(BaseModel):
    """Location information from alarm."""
    address: str
    lat: Optional[float] = None
    lng: Optional[float] = None


class AlarmWebhookPayload(BaseModel):
    """Expected alarm webhook payload."""
    alarm_id: str = Field(..., description="Unique alarm identifier")
    timestamp: str = Field(..., description="ISO8601 timestamp")
    type: str = Field(..., description="Alarm type (fire, medical, technical, etc.)")
    priority: Optional[str] = Field(None, description="Optional priority override")
    location: AlarmLocation
    description: str = Field(..., description="Alarm description/details")
    source: str = Field(default="alarm_webhook", description="Alarm source")


class AlarmWebhookResponse(BaseModel):
    """Response from alarm webhook."""
    status: str  # "success" or "error"
    incident_id: Optional[str] = None
    message: str


@router.post("/webhook", response_model=AlarmWebhookResponse)
async def receive_alarm_webhook(
    request: Request,
    payload: AlarmWebhookPayload,
    x_webhook_signature: str = Header(..., alias="X-Webhook-Signature"),
    db: AsyncSession = Depends(get_db)
):
    """
    Receive alarm webhook from alarm server.

    **Authentication**: HMAC-SHA256 signature in X-Webhook-Signature header

    **Expected payload**:
    ```json
    {
        "alarm_id": "ALM-2024-001",
        "timestamp": "2024-10-24T14:30:00Z",
        "type": "fire",
        "priority": "high",
        "location": {
            "address": "Hauptstrasse 10, 4104 Oberwil",
            "lat": 47.5149,
            "lng": 7.5177
        },
        "description": "Brand in Wohnhaus, Rauchentwicklung",
        "source": "divera"
    }
    ```

    **Response**:
    ```json
    {
        "status": "success",
        "incident_id": "uuid-here",
        "message": "Incident created successfully"
    }
    ```
    """
    service = get_alarm_webhook_service()

    # Verify signature
    body = await request.body()
    if not service.verify_signature(body, x_webhook_signature):
        logger.error("Invalid webhook signature")
        raise HTTPException(status_code=403, detail="Invalid signature")

    logger.info(f"Received alarm webhook: alarm_id={payload.alarm_id}")

    try:
        # Process alarm and create incident
        incident = await service.process_alarm(db, payload.dict())

        return AlarmWebhookResponse(
            status="success",
            incident_id=str(incident.id),
            message="Incident created successfully"
        )

    except ValueError as e:
        logger.error(f"Alarm processing failed: {str(e)}")
        return AlarmWebhookResponse(
            status="error",
            message=str(e)
        )

    except Exception as e:
        logger.error(f"Unexpected error processing alarm: {str(e)}", exc_info=True)
        raise HTTPException(
            status_code=500,
            detail="Failed to process alarm"
        )

"""
Test script for alarm webhook.

Usage:
    python -m app.scripts.test_alarm_webhook --secret YOUR_SECRET
"""

import argparse
import json
import hmac
import hashlib
import requests
from datetime import datetime


def generate_signature(payload: bytes, secret: str) -> str:
    """Generate HMAC-SHA256 signature."""
    return hmac.new(
        secret.encode(),
        payload,
        hashlib.sha256
    ).hexdigest()


def send_test_alarm(api_url: str, secret: str):
    """Send test alarm to webhook."""
    payload = {
        "alarm_id": f"TEST-{datetime.now().strftime('%Y%m%d-%H%M%S')}",
        "timestamp": datetime.now().isoformat(),
        "type": "fire",
        "priority": "high",
        "location": {
            "address": "Hauptstrasse 10, 4104 Oberwil",
            "lat": 47.5149,
            "lng": 7.5177
        },
        "description": "Test alarm - Brand in Wohnhaus, Rauchentwicklung",
        "source": "test_script"
    }

    payload_json = json.dumps(payload)
    signature = generate_signature(payload_json.encode(), secret)

    print(f"Sending test alarm to: {api_url}")
    print(f"Payload: {payload_json}")
    print(f"Signature: {signature}")

    response = requests.post(
        f"{api_url}/api/alarms/webhook",
        json=payload,
        headers={
            "X-Webhook-Signature": signature,
            "Content-Type": "application/json"
        }
    )

    print(f"\nResponse status: {response.status_code}")
    print(f"Response body: {response.json()}")

    return response


def main():
    parser = argparse.ArgumentParser(description="Test alarm webhook")
    parser.add_argument(
        "--url",
        default="http://localhost:8000",
        help="API base URL"
    )
    parser.add_argument(
        "--secret",
        required=True,
        help="Webhook secret"
    )

    args = parser.parse_args()

    send_test_alarm(args.url, args.secret)


if __name__ == "__main__":
    main()

## Security Considerations

### Webhook Authentication
- HMAC-SHA256 signature verification required
- Shared secret stored securely in environment variables
- Constant-time signature comparison prevents timing attacks
- Invalid signatures logged but don't reveal details

### Input Validation
- All payload fields validated with Pydantic
- SQL injection prevented by ORM
- XSS prevention in description field
- Rate limiting on webhook endpoint (consider adding)

### Error Handling
- Failed alarms logged but don't expose system details
- Graceful degradation if geocoding fails
- Duplicate detection prevents spam

## Performance Notes

### Optimization Strategies
- **Async processing**: Webhook handled asynchronously
- **Duplicate detection**: Simple query with time window
- **Geocoding cache**: Reuses cached results
- **Database indexing**: Index on location_address for duplicates

### Monitoring
- Log all incoming alarms
- Track processing time
- Alert on repeated failures
- Monitor duplicate detection rate
