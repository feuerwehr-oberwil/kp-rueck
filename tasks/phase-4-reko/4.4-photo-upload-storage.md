```python
"""
Photo storage service with compression and filesystem management.
"""

import os
import uuid
import logging
from pathlib import Path
from typing import Optional, Tuple
from datetime import datetime, timedelta
from PIL import Image
import io

from app.config import get_settings

logger = logging.getLogger(__name__)


class PhotoStorageService:
    """Service for managing photo uploads and storage."""

    def __init__(self):
        self.settings = get_settings()
        self.storage_path = Path(self.settings.PHOTO_STORAGE_PATH)
        self.max_file_size = 10 * 1024 * 1024  # 10MB
        self.allowed_extensions = {'.jpg', '.jpeg', '.png', '.webp', '.heic'}
        self.target_format = 'JPEG'
        self.target_quality = 85
        self.max_dimension = 1920  # Max width or height

        # Ensure storage directory exists
        self.storage_path.mkdir(parents=True, exist_ok=True)
        logger.info(f"Photo storage initialized at: {self.storage_path}")

    def _validate_file(self, file_data: bytes, filename: str) -> Tuple[bool, Optional[str]]:
        """
        Validate uploaded file.

        Returns:
            (is_valid, error_message)
        """
        # Check file size
        if len(file_data) > self.max_file_size:
            return False, f"File too large: {len(file_data)} bytes (max {self.max_file_size})"

        # Check extension
        ext = Path(filename).suffix.lower()
        if ext not in self.allowed_extensions:
            return False, f"Invalid file type: {ext} (allowed: {self.allowed_extensions})"

        # Try to open as image
        try:
            with Image.open(io.BytesIO(file_data)) as img:
                img.verify()
        except Exception as e:
            return False, f"Invalid image file: {str(e)}"

        return True, None

    def _compress_image(self, file_data: bytes) -> Tuple[bytes, int, int]:
        """
        Compress and resize image.

        Returns:
            (compressed_data, width, height)
        """
        with Image.open(io.BytesIO(file_data)) as img:
            # Convert RGBA to RGB (for PNG with transparency)
            if img.mode in ('RGBA', 'LA', 'P'):
                background = Image.new('RGB', img.size, (255, 255, 255))
                if img.mode == 'P':
                    img = img.convert('RGBA')
                background.paste(img, mask=img.split()[-1] if img.mode in ('RGBA', 'LA') else None)
                img = background

            # Resize if too large
            if img.width > self.max_dimension or img.height > self.max_dimension:
                img.thumbnail((self.max_dimension, self.max_dimension), Image.Resampling.LANCZOS)
                logger.info(f"Resized image to {img.width}x{img.height}")

            # Save to bytes with compression
            output = io.BytesIO()
            img.save(output, format=self.target_format, quality=self.target_quality, optimize=True)
            compressed_data = output.getvalue()

            return compressed_data, img.width, img.height

    def _generate_filename(self, incident_id: str, original_filename: str) -> str:
        """
        Generate unique filename for photo.

        Format: {incident_id}_{uuid}_{timestamp}.jpg
        """
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        unique_id = str(uuid.uuid4())[:8]
        return f"{incident_id}_{unique_id}_{timestamp}.jpg"

    def _get_incident_directory(self, incident_id: str) -> Path:
        """Get or create directory for incident photos."""
        incident_dir = self.storage_path / incident_id
        incident_dir.mkdir(parents=True, exist_ok=True)
        return incident_dir

    async def save_photo(
        self,
        incident_id: str,
        file_data: bytes,
        original_filename: str
    ) -> Tuple[str, dict]:
        """
        Save photo with compression.

        Args:
            incident_id: ID of the incident
            file_data: Raw file bytes
            original_filename: Original filename

        Returns:
            (photo_filename, metadata_dict)
        """
        # Validate file
        is_valid, error_msg = self._validate_file(file_data, original_filename)
        if not is_valid:
            raise ValueError(error_msg)

        # Compress image
        original_size = len(file_data)
        compressed_data, width, height = self._compress_image(file_data)
        compressed_size = len(compressed_data)

        logger.info(
            f"Compressed image: {original_size} -> {compressed_size} bytes "
            f"({100 * (1 - compressed_size / original_size):.1f}% reduction)"
        )

        # Generate filename and save
        filename = self._generate_filename(incident_id, original_filename)
        incident_dir = self._get_incident_directory(incident_id)
        file_path = incident_dir / filename

        with open(file_path, 'wb') as f:
            f.write(compressed_data)

        logger.info(f"Saved photo: {file_path}")

        metadata = {
            "filename": filename,
            "original_filename": original_filename,
            "original_size_bytes": original_size,
            "compressed_size_bytes": compressed_size,
            "width": width,
            "height": height,
            "format": self.target_format,
            "uploaded_at": datetime.now().isoformat()
        }

        return filename, metadata

    def get_photo_path(self, incident_id: str, filename: str) -> Optional[Path]:
        """
        Get full path to photo file.

        Returns None if file doesn't exist.
        """
        file_path = self.storage_path / incident_id / filename
        return file_path if file_path.exists() else None

    def delete_photo(self, incident_id: str, filename: str) -> bool:
        """
        Delete photo file.

        Returns True if deleted, False if not found.
        """
        file_path = self.storage_path / incident_id / filename
        if file_path.exists():
            file_path.unlink()
            logger.info(f"Deleted photo: {file_path}")
            return True
        return False

    def delete_incident_photos(self, incident_id: str) -> int:
        """
        Delete all photos for an incident.

        Returns number of deleted files.
        """
        incident_dir = self.storage_path / incident_id
        if not incident_dir.exists():
            return 0

        count = 0
        for file_path in incident_dir.iterdir():
            if file_path.is_file():
                file_path.unlink()
                count += 1

        # Remove directory if empty
        if not any(incident_dir.iterdir()):
            incident_dir.rmdir()
            logger.info(f"Removed empty directory: {incident_dir}")

        logger.info(f"Deleted {count} photos for incident {incident_id}")
        return count

    def get_storage_stats(self) -> dict:
        """
        Get storage statistics.

        Returns:
            Dict with total_photos, total_size_bytes, incidents_count
        """
        total_photos = 0
        total_size = 0
        incidents = set()

        for incident_dir in self.storage_path.iterdir():
            if incident_dir.is_dir():
                incidents.add(incident_dir.name)
                for file_path in incident_dir.iterdir():
                    if file_path.is_file():
                        total_photos += 1
                        total_size += file_path.stat().st_size

        return {
            "total_photos": total_photos,
            "total_size_bytes": total_size,
            "total_size_mb": round(total_size / (1024 * 1024), 2),
            "incidents_count": len(incidents),
            "storage_path": str(self.storage_path)
        }

    def cleanup_old_photos(self, days: int = 365) -> int:
        """
        Delete photos older than specified days.

        Returns number of deleted files.
        """
        cutoff_date = datetime.now() - timedelta(days=days)
        deleted_count = 0

        for incident_dir in self.storage_path.iterdir():
            if not incident_dir.is_dir():
                continue

            for file_path in incident_dir.iterdir():
                if not file_path.is_file():
                    continue

                # Check file modification time
                mtime = datetime.fromtimestamp(file_path.stat().st_mtime)
                if mtime < cutoff_date:
                    file_path.unlink()
                    deleted_count += 1
                    logger.info(f"Deleted old photo: {file_path}")

            # Remove directory if empty
            if not any(incident_dir.iterdir()):
                incident_dir.rmdir()

        logger.info(f"Cleanup: deleted {deleted_count} photos older than {days} days")
        return deleted_count


# Global service instance
_photo_storage: Optional[PhotoStorageService] = None


def get_photo_storage() -> PhotoStorageService:
    """Get or create the global photo storage service instance."""
    global _photo_storage
    if _photo_storage is None:
        _photo_storage = PhotoStorageService()
    return _photo_storage
```

**File**: `backend/app/api/reko.py` (update photo upload)

```python
from fastapi import UploadFile, File
from fastapi.responses import FileResponse
from app.services.photo_storage import get_photo_storage

@router.post("/{incident_id}/photos")
async def upload_reko_photo(
    incident_id: str,
    photo: UploadFile = File(...),
    token: str = Header(..., alias="X-Reko-Token"),
    db: AsyncSession = Depends(get_db)
):
    """
    Upload photo for Reko report.

    Supports: JPEG, PNG, WebP, HEIC
    Max size: 10MB
    Automatic compression and format conversion to JPEG
    """
    # Validate token
    is_valid = await validate_reko_token(db, incident_id, token)
    if not is_valid:
        raise HTTPException(status_code=403, detail="Invalid token")

    # Read file data
    file_data = await photo.read()

    # Save with compression
    try:
        photo_storage = get_photo_storage()
        filename, metadata = await photo_storage.save_photo(
            incident_id,
            file_data,
            photo.filename or "photo.jpg"
        )

        logger.info(
            f"Photo uploaded: {filename} "
            f"({metadata['original_size_bytes']} -> {metadata['compressed_size_bytes']} bytes)"
        )

        # Return photo URL
        photo_url = f"/api/photos/{incident_id}/{filename}"

        return {
            "photo_url": photo_url,
            "filename": filename,
            "metadata": metadata
        }

    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except Exception as e:
        logger.error(f"Photo upload failed: {str(e)}")
        raise HTTPException(status_code=500, detail="Photo upload failed")


@router.get("/photos/{incident_id}/{filename}")
async def serve_photo(
    incident_id: str,
    filename: str
):
    """
    Serve photo file.

    Returns image with appropriate cache headers.
    """
    photo_storage = get_photo_storage()
    file_path = photo_storage.get_photo_path(incident_id, filename)

    if not file_path:
        raise HTTPException(status_code=404, detail="Photo not found")

    # Serve file with cache headers
    return FileResponse(
        path=file_path,
        media_type="image/jpeg",
        headers={
            "Cache-Control": "public, max-age=31536000",  # 1 year
            "ETag": f'"{filename}"'
        }
    )


@router.delete("/{incident_id}/photos/{filename}")
async def delete_reko_photo(
    incident_id: str,
    filename: str,
    token: str = Header(..., alias="X-Reko-Token"),
    db: AsyncSession = Depends(get_db)
):
    """Delete photo from Reko report."""
    # Validate token
    is_valid = await validate_reko_token(db, incident_id, token)
    if not is_valid:
        raise HTTPException(status_code=403, detail="Invalid token")

    photo_storage = get_photo_storage()
    deleted = photo_storage.delete_photo(incident_id, filename)

    if not deleted:
        raise HTTPException(status_code=404, detail="Photo not found")

    return {"message": "Photo deleted"}
```

### 7. Add Backup Script

**File**: `backend/scripts/backup_photos.sh`

```bash
#!/bin/bash
# Backup photos to external storage

set -e

BACKUP_DIR="${BACKUP_DIR:-/backups/photos}"
STORAGE_DIR="${PHOTO_STORAGE_PATH:-/app/data/photos}"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="${BACKUP_DIR}/photos_${TIMESTAMP}.tar.gz"

echo "Starting photo backup..."
echo "Source: ${STORAGE_DIR}"
echo "Destination: ${BACKUP_FILE}"

# Create backup directory
mkdir -p "${BACKUP_DIR}"

# Create compressed archive
tar -czf "${BACKUP_FILE}" -C "$(dirname ${STORAGE_DIR})" "$(basename ${STORAGE_DIR})"

echo "Backup complete: ${BACKUP_FILE}"

# Get file size
SIZE=$(du -h "${BACKUP_FILE}" | cut -f1)
echo "Backup size: ${SIZE}"

# Optional: Upload to S3 or external storage
# aws s3 cp "${BACKUP_FILE}" s3://your-bucket/photos/

# Cleanup old backups (keep last 30 days)
find "${BACKUP_DIR}" -name "photos_*.tar.gz" -mtime +30 -delete

echo "Backup finished successfully"
```
